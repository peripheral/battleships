package app.logics.board.ship.test;

import static org.junit.jupiter.api.Assertions.assertFalse;
import static org.junit.jupiter.api.Assertions.assertTrue;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import app.logics.board.ship.Ship;
import app.logics.board.ship.Ship.ORIENTATION;
import app.logics.board.ship.Submarine;

/**
 * Performs white box testing in Ship class
 * @author Artur Vitt
 *
 */
public class ShipTest {
	private Ship sut;
	
	@BeforeEach
	void init() {
		sut = new Submarine();
	}	
	
	@Test
	void testContainsShouldReturnTrue() {
		int locX = 0;
		int locY = 0;
		sut.setOrientation(ORIENTATION.VERTICAL);
		sut.setStartLocation(new int[] {locX,locY});
		assertTrue(sut.contains(locX, locY));
	}
	
	@Test
	void testContainsShouldReturnFalse() {
		int locX = 0;
		int locY = 0;
		int loc2Y = 3;
		sut.setOrientation(ORIENTATION.VERTICAL);
		sut.setStartLocation(new int[] {locX,locY});
		assertFalse(sut.contains(locX, loc2Y));
	}
}
